from abc import ABC, abstractmethod

class Strategy(ABC):
    @abstractmethod
    def execute(self, a, b):
        pass

class Add(Strategy):
    def execute(self, a, b):
        return a + b

class Multiply(Strategy):
    def execute(self, a, b):
        return a * b

class Calculator:
    def __init__(self, strategy: Strategy):
        self.strategy = strategy

    def calculate(self, a, b):
        return self.strategy.execute(a, b)

calc = Calculator(Add())
print(calc.calculate(5, 3))  # 8

calc.strategy = Multiply()
print(calc.calculate(5, 3))  # 15

class Publisher:
    def __init__(self):
        self.subscribers = []

    def subscribe(self, subscriber): self.subscribers.append(subscriber)
    def notify(self, data):
        for sub in self.subscribers:
            sub.update(data)

class Subscriber:
    def update(self, data): print(f"Received: {data}")

sub1 = Subscriber()
